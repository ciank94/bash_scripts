;;Aliases based on functions below
(defalias 'fimp 'impnone)
(defalias 'frea 'real)
(defalias 'fint 'int)

;;Functions for defining the emacs aliases
(defun impnone ()
  (interactive)
  (insert "implicit none")
    (backward-char 2))

(defun real ()
  (interactive)
  (insert "real,dimension(:),allocatable::")
  (backward-char 2))

(defun int ()
  (interactive)
  (insert "integer::")
  (backward-char 2))
